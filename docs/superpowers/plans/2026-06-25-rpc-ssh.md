# RPC-Backed `yeet ssh` Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make host and non-VM service `yeet ssh` sessions use catch RPC over tsnet with the existing WebSocket PTY transport, while VM targets keep guest SSH.

**Architecture:** Add explicit catch RPC exec targets for host shell and service shell. Preserve the existing empty-target service command dispatcher for all normal remote yeet commands. Route `yeet ssh` host and non-VM service targets to the RPC shell path, and leave VM services on the current OpenSSH guest path.

**Tech Stack:** Go, `catchrpc.ExecRequest`, Gorilla WebSocket, catch `ttyExecer`, existing `pkg/yeet` remote exec terminal handling, website docs submodule.

---

## File Structure

- `pkg/catchrpc/types.go`: add `ExecTarget` constants and `ExecRequest.Target`.
- `pkg/catch/rpc.go`: validate exec targets before starting `ttyExecer`; pass target into the executor.
- `pkg/catch/tty_exec.go`: add host-shell and service-shell execution paths using existing PTY and command helpers.
- `pkg/catch/tty_test.go`: cover shell command execution, shell resolution, service cwd behavior, and VM rejection.
- `pkg/catch/rpc_test.go`: cover exec target validation.
- `pkg/yeet/exec_remote.go`: add a reusable RPC shell helper that preserves raw mode, resize, stdin proxying, output tracking, and remote exit handling.
- `pkg/yeet/exec_remote_test.go`: cover RPC shell request construction.
- `pkg/yeet/ssh_cmd.go`: route host and non-VM service `yeet ssh` through RPC shell; keep VM SSH.
- `pkg/yeet/ssh_cmd_test.go`: cover `yeet ssh` routing decisions and option validation.
- `cmd/yeet/cli.go` and `cmd/yeet/cli_test.go`: update user-facing CLI help.
- `README.md` and `website/docs/cli/yeet-cli.mdx`: update user docs for catch RPC shell behavior.

func normalizeExecRequest(req catchrpc.ExecRequest) (catchrpc.ExecRequest, error) {
	switch req.Target {
	case catchrpc.ExecTargetServiceCommand:
		if strings.TrimSpace(req.Service) == "" {
			return catchrpc.ExecRequest{}, fmt.Errorf("missing service")
		}
		return req, nil
	case catchrpc.ExecTargetHostShell:
		req.Service = ""
		return req, nil
	case catchrpc.ExecTargetServiceShell:
		if strings.TrimSpace(req.Service) == "" {
			return catchrpc.ExecRequest{}, fmt.Errorf("missing service")
		}
		return req, nil
	default:
		return catchrpc.ExecRequest{}, fmt.Errorf("unknown exec target %q", req.Target)
	}
}

func (e *ttyExecer) serviceShellDir() (string, error) {
	st, err := e.s.serviceType(e.sn)
	if err != nil {
		return "", fmt.Errorf("failed to get service type: %w", err)
	}
	if st == db.ServiceTypeVM {
		return "", fmt.Errorf("service %q is a VM service; VM targets use guest SSH", e.sn)
	}
	root, err := e.s.serviceRootDir(e.sn)
	if err != nil {
		return "", err
	}
	return serviceDataDirForRoot(root), nil
}

func (e *ttyExecer) runShellCommand(args []string, dir string) error {
	var cmd *exec.Cmd
	if len(args) == 0 {
		shell := e.defaultShellPath()
		cmd = e.newCmd(shell)
		cmd.Args[0] = "-" + filepath.Base(shell)
	} else {
		cmd = e.newCmd(args[0], args[1:]...)
	}
	if dir != "" {
		cmd.Dir = dir
	}
	return cmd.Run()
}

func (e *ttyExecer) defaultShellPath() string {
	if user := strings.TrimSpace(e.s.cfg.InstallUser); user != "" {
		if shell, ok := loginShellForUser(user); ok {
			return shell
		}
	}
	if shell, ok := loginShellForUser("root"); ok {
		return shell
	}
	return "/bin/sh"
}

func loginShellForUser(username string) (string, bool) {
	raw, err := os.ReadFile(passwdFilePath)
	if err != nil {
		return "", false
	}
	for _, line := range strings.Split(string(raw), "\n") {
		if line == "" || strings.HasPrefix(line, "#") {
			continue
		}
		fields := strings.Split(line, ":")
		if len(fields) < 7 || fields[0] != username {
			continue
		}
		shell := strings.TrimSpace(fields[6])
		if shell == "" {
			return "", false
		}
		return shell, true
	}
	return "", false
}

func newRemoteShellExecRequest(host string, target catchrpc.ExecTarget, service string, args []string, stdin io.Reader, tty bool) catchrpc.ExecRequest {
	req := newRemoteExecRequest(host, service, args, stdin, tty)
	req.Target = target
	if target == catchrpc.ExecTargetHostShell {
		req.Service = ""
	}
	return req
}

func execRemoteShell(ctx context.Context, host string, target catchrpc.ExecTarget, service string, args []string, stdin io.Reader, tty bool, stdout io.Writer) (err error) {
	if stdout == nil {
		stdout = io.Discard
	}
	client := newRPCExecClientFn(host)
	session := &remoteExecSession{
		req:   newRemoteShellExecRequest(host, target, service, args, stdin, applyTTYOverride(tty)),
		stdin: stdin,
	}
	session.configureTTY(ctx)
	if err := session.prepareTTYStdin(); err != nil {
		session.close(&err)
		return err
	}
	defer session.close(&err)

	out := &trackingWriter{w: stdout}
	code, execErr := client.Exec(ctx, session.req, session.stdin, out, session.resizeCh)
	if execErr != nil {
		return execErr
	}
	return remoteExecExitError(code, session.rawMode, out)
}
