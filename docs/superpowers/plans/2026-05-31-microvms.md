# Experimental VM Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add experimental `vm://ubuntu/26.04` support so `yeet run <name> vm://ubuntu/26.04` creates a long-lived Firecracker-backed Ubuntu VM with `type = "vm"`, sane host-based defaults, service-style lifecycle, console access, and SSH target selection.

**Architecture:** Treat a VM as a service type named `vm`. The client recognizes `vm://` as a fileless run payload, saves it in `yeet.toml`, and forwards it to catch as a remote `run` payload argument. Catch owns VM provisioning, image cache, storage, Firecracker config, systemd lifecycle, networking, service info, and console bridging.

**Tech Stack:** Go, existing yeet RPC/TTY bridge, JSON DB with generated views/cloners, systemd, Firecracker, qemu-img, ZFS zvols, Linux TAP/veth/macvtap networking, Ubuntu 26.04 image bundle hosted from GitHub.

---

## Scope

This plan implements the first usable v0. It keeps the guest image owned by yeet, keeps arbitrary rootfs/kernel support out of scope, and makes the public type string `vm`.

func isVMPayload(payload string) bool {
	return strings.TrimSpace(payload) == vmUbuntu2604Payload

func normalizeRunDraftPayload(cwd, payload, kind string) (string, string, error) {
	payload = strings.TrimSpace(payload)
	if isVMPayload(payload) {
		if kind != "" && kind != "auto" && kind != serviceTypeVM {
			return "", kind, fmt.Errorf("payloadKind %q is not valid for VM payloads", kind)
		}
		return payload, serviceTypeVM, nil
	}
	switch kind {
	case "", "auto":
		return normalizeAutoRunDraftPayload(cwd, payload)
	case serviceTypeVM:
		return "", kind, fmt.Errorf("payloadKind %q requires %s", serviceTypeVM, vmUbuntu2604Payload)
	case "file":
		normalized, err := normalizeExistingRunDraftPath(cwd, payload)
		return normalized, kind, err
	case "compose":
		return normalizeRunDraftComposePayload(cwd, payload)
	case "dockerfile":
		return normalizeRunDraftDockerfilePayload(cwd, payload)
	case "remote-image":
		if !looksLikeRunDraftImageRef(payload) {
			return "", kind, fmt.Errorf("payload must be a Docker image reference for payloadKind %q", kind)
		}
		return payload, kind, nil
	case "local-image":
		if !looksLikeRunDraftImageRef(payload) && !looksLikeRunDraftLocalImageName(payload) {
			return "", kind, fmt.Errorf("payload must be a Docker image reference or local image name for payloadKind %q", kind)
		}
		return payload, kind, nil
	}
	return "", kind, fmt.Errorf("unknown payload kind %q", kind)
}

func unknownPayloadKind(kind string) bool {
	switch kind {
	case "", "auto", "file", "compose", "dockerfile", "remote-image", "local-image", serviceTypeVM:
		return false
	default:
		return true
	}
}

func parseVMSize(raw string) (int64, error) {
	raw = strings.TrimSpace(strings.ToLower(raw))
	if raw == "" {
		return 0, nil
	}
	mult := int64(1)
	switch {
	case strings.HasSuffix(raw, "g"):
		mult = 1 << 30
		raw = strings.TrimSuffix(raw, "g")
	case strings.HasSuffix(raw, "gb"):
		mult = 1 << 30
		raw = strings.TrimSuffix(raw, "gb")
	case strings.HasSuffix(raw, "m"):
		mult = 1 << 20
		raw = strings.TrimSuffix(raw, "m")
	case strings.HasSuffix(raw, "mb"):
		mult = 1 << 20
		raw = strings.TrimSuffix(raw, "mb")
	}
	n, err := strconv.ParseInt(strings.TrimSpace(raw), 10, 64)
	if err != nil || n <= 0 {
		return 0, fmt.Errorf("invalid size %q", raw)
	}
	return n * mult, nil
}

func validateVMHost(profile vmHostProfile) error {
	if !profile.HasKVM {
		return fmt.Errorf("VMs require Linux KVM; /dev/kvm is missing")
	}
	if profile.Arch != "x86_64" && profile.Arch != "amd64" {
		return fmt.Errorf("VMs require x86_64/amd64 in v0; got %s", profile.Arch)
	}
	return nil
}

func defaultVMShape(profile vmHostProfile) (vmShape, error) {
	if err := validateVMHost(profile); err != nil {
		return vmShape{}, err
	}
	shape := vmShape{
		CPUs:        defaultVMCPUs(profile.LogicalCPUs),
		MemoryBytes: defaultVMMemory(profile.MemoryBytes),
		DiskBytes:   defaultVMDisk(profile.StorageBytes, profile.StorageZFS),
		DiskBackend: vmDiskBackendRaw,
	}
	if profile.StorageZFS {
		shape.DiskBackend = vmDiskBackendZVOL
	}
	if shape.DiskBytes == 0 {
		return vmShape{}, fmt.Errorf("not enough storage for VM disk")
	}
	if err := admitVMMemory(profile, shape.MemoryBytes); err != nil {
		return vmShape{}, err
	}
	return shape, nil
}

func defaultVMCPUs(hostCPUs int) int {
	switch {
	case hostCPUs <= 1:
		return 1
	case hostCPUs < 8:
		return 2
	default:
		return 4
	}
}

func defaultVMMemory(hostBytes int64) int64 {
	switch {
	case hostBytes <= 2<<30:
		return 512 << 20
	case hostBytes <= 4<<30:
		return 1 << 30
	case hostBytes < 16<<30:
		return 2 << 30
	default:
		return 4 << 30
	}
}

func defaultVMDisk(availableBytes int64, zfs bool) int64 {
	if zfs {
		switch {
		case availableBytes >= 512<<30:
			return 128 << 30
		case availableBytes >= 128<<30:
			return 64 << 30
		case availableBytes >= 64<<30:
			return 32 << 30
		case availableBytes >= 32<<30:
			return 16 << 30
		case availableBytes >= 12<<30:
			return 8 << 30
		default:
			return 0
		}
	}
	switch {
	case availableBytes >= 48<<30:
		return 32 << 30
	case availableBytes >= 24<<30:
		return 16 << 30
	case availableBytes >= 12<<30:
		return 8 << 30
	default:
		return 0
	}
}

func admitVMMemory(profile vmHostProfile, requestBytes int64) error {
	reserve := vmHostMemoryReserve(profile.MemoryBytes)
	budget := profile.MemoryBytes - reserve - profile.RunningVMBytes
	if requestBytes > budget {
		return fmt.Errorf("not enough memory to start VM: requested %s, available budget %s", formatBytesInt(requestBytes), formatBytesInt(budget))
	}
	return nil
}

	tenPercent := total / 10
	if tenPercent < 1<<30 {
		return 1 << 30
	}
	if tenPercent > 4<<30 {
		return 4 << 30
	}
	return tenPercent
}

	Version      string            `json:"version"`
	Architecture string           `json:"architecture"`
	Kernel       string           `json:"kernel"`
	RootFS       string           `json:"rootfs"`

	Manifest      vmImageManifest

	KernelPath    string
	RootFSPath    string
	FirecrackerPath string

func (c vmImageCache) Ensure(ctx context.Context) (vmImagePaths, error) {
	manifest, err := c.fetchManifest(ctx)
	if err != nil {
		return vmImagePaths{}, err
	}
	dir := filepath.Join(c.Root, manifest.Version)
	if err := os.MkdirAll(dir, 0o755); err != nil {
		return vmImagePaths{}, err
	}
	if err := c.ensureArtifact(ctx, dir, manifest, manifest.Kernel); err != nil {
		return vmImagePaths{}, err
	}
	if err := c.ensureArtifact(ctx, dir, manifest, manifest.RootFS); err != nil {
		return vmImagePaths{}, err
	}
	if err := c.ensureArtifact(ctx, dir, manifest, manifest.Firecracker); err != nil {
		return vmImagePaths{}, err
	}
	fc := filepath.Join(dir, manifest.Firecracker)
	if err := os.Chmod(fc, 0o755); err != nil {
		return vmImagePaths{}, err
	}
	return vmImagePaths{
		Manifest:        manifest,
		Dir:             dir,
		KernelPath:      filepath.Join(dir, manifest.Kernel),
		RootFSPath:      filepath.Join(dir, manifest.RootFS),

	}

	snap := p.BaseDataset + "@" + p.ImageVersion
	size := fmt.Sprintf("%d", p.Bytes)
	return [][]string{
		{"zfs", "create", "-s", "-V", size, p.BaseDataset},
		{"dd", "if=" + p.BaseRootFS, "of=/dev/zvol/" + p.BaseDataset, "bs=16M", "status=none"},

	}

	}

		return err
	}

		return err
	}
	return os.WriteFile(filepath.Join(dir, "network.yaml"), []byte(renderVMNetworkYAML(cfg.Networks)), 0o644)
}

func renderVMNetworkYAML(networks []vmGuestNetwork) string {
	var b strings.Builder

			continue
		}
		fmt.Fprintf(&b, "      addresses: [%s]\n", net.Address)
		if net.Gateway != "" {
			fmt.Fprintf(&b, "      gateway4: %s\n", net.Gateway)
		}
	}
	return b.String()

	Parent    string
	MAC       string
	GuestIP   string
	Gateway   string
	DHCP      bool

func newVMNetworkPlan(service string, modes []string, in vmNetworkInputs) vmNetworkPlan {
	plan := vmNetworkPlan{Service: service}
	for idx, mode := range modes {
		switch mode {
		case "svc":
			plan.Interfaces = append(plan.Interfaces, vmNetworkInterfacePlan{
				Mode:      "svc",

func (e *ttyExecer) vmConsoleCmdFunc() error {
	sv, err := e.s.serviceView(e.sn)
	if err != nil {
		return err
	}
	if sv.ServiceType() != db.ServiceTypeVM {
		return fmt.Errorf("service %q is type %q; vm console requires type \"vm\"", e.sn, sv.ServiceType())
	}
	vm := sv.AsStruct().VM
	if vm == nil || strings.TrimSpace(vm.Console.SocketPath) == "" {
		return fmt.Errorf("service %q has no VM console socket", e.sn)
	}
	conn, err := net.Dial("unix", vm.Console.SocketPath)
	if err != nil {
		return fmt.Errorf("connect VM console: %w", err)
	}
	defer conn.Close()
	errc := make(chan error, 2)
	go func() { _, err := io.Copy(conn, e.rw); errc <- err }()
	go func() { _, err := io.Copy(e.rw, conn); errc <- err }()
	err = <-errc
	if isExpectedCopyErr(err) {
		return nil
	}
	return err
}
